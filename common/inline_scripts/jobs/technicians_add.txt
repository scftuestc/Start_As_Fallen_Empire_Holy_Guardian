# Regular Empires
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_regular_empire = yes
			is_fallen_empire_spiritualist = no
		}
	}
	modifier = {
		job_technician_add = $AMOUNT$
	}
}

# Spiritualist FEs
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_fallen_empire_spiritualist = yes
			is_ai = yes
		}
	}
	modifier = {
		job_fe_acolyte_generator_add = $AMOUNT$
	}
}
