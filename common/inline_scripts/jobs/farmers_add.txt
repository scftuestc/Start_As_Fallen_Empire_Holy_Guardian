# Regular Empires
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_regular_empire = yes
			is_fallen_empire_spiritualist = no
			is_gestalt = no
		}
	}
	modifier = {
		job_farmer_add = $AMOUNT$
	}
}
