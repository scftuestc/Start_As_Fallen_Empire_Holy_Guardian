d_dust_caverns = {
	is_for_colonizable = yes
	category = deposit_cat_rare_resource

	potential = {
		can_have_habitable_deposits = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_mine_volatile_motes
				NOT = { has_origin = Start_As_Fallen_Empire_Holy_Guardians }
			}
		}
		planet_technician_volatile_motes_produces_add = @SR_SMALL
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_mine_volatile_motes
				has_origin = Start_As_Fallen_Empire_Holy_Guardians
			}
		}
		planet_safe_fe_acolyte_generator_volatile_motes_produces_add = @SR_SMALL
	}

	planet_modifier = {
		district_generator_max_add = @DISTRICTS_FROM_SR_DEPOSITS
	}

	drop_weight = {
		weight = @high_rare
		modifier = {
			factor = @planet_type_bonus
			is_dry = yes
		}
		modifier = {
			factor = @gaia_bonus
			is_planet_class = pc_gaia
		}
		modifier = {
			factor = 0
			solar_system = {
				OR = {
					has_star_flag = empire_home_system
					has_star_flag = sol_system
					has_star_flag = deneb_system
				}
			}
		}
	}
}
